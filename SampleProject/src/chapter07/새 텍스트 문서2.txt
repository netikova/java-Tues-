package chapter07;

import java.util.ArrayList;
import java.util.Scanner;
import java.util.StringTokenizer;

class Student{
	private String name;
	private String sub;
	private int id;
	private double avg;
	public void student(String name,String sub,int id,double avg) {
	this.name = name;
	this.sub = sub;
	this.id = id;
	this.avg = avg;
	}
	@Override
	public String toString() {
		return "Student [name=" + name + ", sub=" + sub + ", id=" + id + ", avg=" + avg + "]";
	}
	
}
