package chapter07;

import java.util.ArrayList;
import java.util.Scanner;
import java.util.StringTokenizer;

class Student{
	private String name;
	private int sub;
	private String id;
	private int avg;
	public void student(String name,int sub,String id,double avg) {
	this.name = name;
	this.sub = sub;
	this.id = id;
	this.avg = avg;
	}
	@Override
	public String toString() {
		return "Student [name=" + name + ", sub=" + sub + ", id=" + id + ", avg=" + avg + "]";
	}
	
}
